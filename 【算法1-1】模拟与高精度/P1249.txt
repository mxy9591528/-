#include <iostream>
#include <algorithm>
#include <string>
#include <cmath>
#include <vector>
#include <set>
using namespace std;

set<int> st;
vector<int> res = { 1 };
int n;

void mul(vector<int>& A, int b)
{
	vector<int> res;
	int t = 0;
	for (int i = 0; i < A.size() || t; i++)
	{
		if (i < A.size()) t += A[i] * b;
		res.push_back(t % 10);
		t /= 10;
	}
	while (res.size() > 1 && !res.back()) res.pop_back();
	A = res;
}

int main()
{
	cin >> n;
	for (int i = 2; ; i++)
	{
		if (n - i >= 0) n -= i, st.insert(i);
		else if (n - i == -1)
		{
			st.insert(i + 1);
			st.erase(2);
			break;
		}
		else
		{
			st.insert(i);
			st.erase(abs(n - i));//把k去掉
			break;
		}
	}
	for (auto x : st) mul(res, x), cout << x << ' ';
	cout<<endl;
	for (int i = res.size() - 1; i >= 0; i--) cout << res[i];
	return 0;
}
